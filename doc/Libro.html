def to_s
        text=""
        
    for i in (0..@authors.size-2)
        text << "#{authors[i]} , " 
    end
    text << "#{authors[authors.size-1]}."
    text << "\n#{@title}."
    if @serie.size==1
        text << "\n(#{@serie[0]})."
    end
     text << "\n#{@editorial}; #{@edicion} Edicion (#{@date})."
    for i in (0..@isbn.size-1)
        text << "\nISBN #{@isbn[i].length}: #{@isbn[i]}."
    end
    text
end
