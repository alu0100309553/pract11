def to_s
    text=""
    for i in (0..author.size-2)
        text << " #{self.author[i][0]}, #{(self.author[i][1])[0]}. &"
    end
    text << " #{self.author[author.size-1][0]}, #{(self.author[self.author.size-1][1])[0]}."
    text << " (#{self.datey}"
    if (self.sufijo>0)
       text << (self.sufijo+96).chr 
    end
    text << "). #{self.title}: "
    text << " #{self.subtitle}."
    text << " (#{self.edicion}) "
    text << " (#{self.volumen})."
    text << " #{self.editor}."
    text
end
